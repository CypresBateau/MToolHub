#### 3.2.1 Embedding 模型选择

使用 `pritamdeka/S-PubMedBert-MS-MARCO`（基于 PubMedBERT 的 sentence-transformer，HuggingFace 直接下载，无需申请）。

备选方案：`NeuML/pubmedbert-base-embeddings`（更轻量）。

```python
# backend/app/router/embedding_model.py

from sentence_transformers import SentenceTransformer
from app.config import settings

class EmbeddingModel:
    """单例加载 Embedding 模型"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance.model = SentenceTransformer(
                settings.EMBEDDING_MODEL_NAME  # "pritamdeka/S-PubMedBert-MS-MARCO"
            )
        return cls._instance
    
    def encode(self, texts: list[str]) -> np.ndarray:
        """将文本列表编码为向量"""
        return self.model.encode(texts, normalize_embeddings=True)
```

#### 3.2.2 索引构建

```python
# backend/scripts/build_index.py
"""
离线脚本：读取 registry JSON → 生成 embedding → 存入 FAISS
每次注册表变更后运行一次
"""

import json
import faiss
import pickle
import numpy as np
from app.router.embedding_model import EmbeddingModel

def build_index_for_category(registry_path: str, output_prefix: str):
    """
    为一类资源构建 FAISS 索引
    
    索引的文本由以下字段拼接：
      "{name}. {description}. {description_zh}. Keywords: {keywords joined}"
    
    这样中英文查询都能匹配到。
    """
    with open(registry_path) as f:
        data = json.load(f)
    
    items = data.get("tools") or data.get("models") or data.get("skills")
    
    # 构建待编码文本
    texts = []
    metadata = []
    for item in items:
        if not item.get("enabled", True):
            continue
        
        text = f"{item['name']}. {item['description']}. {item.get('description_zh', '')}. Keywords: {', '.join(item.get('keywords', []))}"
        texts.append(text)
        metadata.append(item)  # 保存完整元数据供检索后使用
    
    # 编码
    model = EmbeddingModel()
    vectors = model.encode(texts)
    
    # 构建 FAISS 索引（内积，因为已 normalize，等价于余弦相似度）
    dim = vectors.shape[1]
    index = faiss.IndexFlatIP(dim)
    index.add(vectors.astype(np.float32))
    
    # 保存
    faiss.write_index(index, f"data/faiss/{output_prefix}_index.faiss")
    with open(f"data/faiss/{output_prefix}_metadata.pkl", "wb") as f:
        pickle.dump(metadata, f)
    
    print(f"Built index for {len(texts)} items → {output_prefix}")

if __name__ == "__main__":
    build_index_for_category("data/registry/tools.json", "tool")
    build_index_for_category("data/registry/models.json", "model")
    build_index_for_category("data/registry/skills.json", "skill")
```

#### 3.2.3 在线向量检索

```python
# backend/app/router/vector_search.py

import faiss
import pickle
import numpy as np
from app.router.embedding_model import EmbeddingModel
from app.config import settings

class VectorSearchEngine:
    """加载 FAISS 索引，提供语义检索"""
    
    def __init__(self):
        self.indices = {}
        self.metadata = {}
        self._load_all()
    
    def _load_all(self):
        for category in ["tool", "model", "skill"]:
            index_path = f"{settings.FAISS_DIR}/{category}_index.faiss"
            meta_path = f"{settings.FAISS_DIR}/{category}_metadata.pkl"
            self.indices[category] = faiss.read_index(index_path)
            with open(meta_path, "rb") as f:
                self.metadata[category] = pickle.load(f)
    
    def search(self, query: str, top_k: int = 5, categories: list[str] = None) -> list[dict]:
        """
        跨类别语义搜索
        
        返回: [{"item": {...元数据}, "score": 0.87, "category": "tool"}, ...]
        """
        if categories is None:
            categories = ["tool", "model", "skill"]
        
        model = EmbeddingModel()
        query_vec = model.encode([query]).astype(np.float32)
        
        results = []
        for cat in categories:
            if cat not in self.indices:
                continue
            scores, indices = self.indices[cat].search(query_vec, top_k)
            for score, idx in zip(scores[0], indices[0]):
                if idx == -1:
                    continue
                results.append({
                    "item": self.metadata[cat][idx],
                    "score": float(score),
                    "category": cat
                })
        
        # 按分数排序，取 top_k
        results.sort(key=lambda x: x["score"], reverse=True)
        return results[:top_k]
```

#### 3.2.4 路由决策

```python
# backend/app/router/route_decision.py

from app.router.vector_search import VectorSearchEngine
from app.schemas.chat import RoutingPlan

class RouteDecisionMaker:
    """
    根据向量检索结果，生成调用计划
    
    策略：
    1. 如果 top1 score > 0.85 → 高置信度，直接选定
    2. 如果 top1 score 在 0.6~0.85 → 中置信度，取 top3 让 Claude 最终决定
    3. 如果 top1 score < 0.6 → 低置信度，走纯对话模式（不调用工具）
    4. 如果涉及图像输入 → 强制包含对应的 model 类资源
    """
    
    THRESHOLD_HIGH = 0.85
    THRESHOLD_LOW = 0.60
    
    def __init__(self, search_engine: VectorSearchEngine):
        self.search = search_engine
    
    def decide(self, user_message: str, has_file: bool = False, file_type: str = None) -> RoutingPlan:
        """
        输入：用户消息 + 是否有附件
        输出：RoutingPlan（包含选中的工具/模型/技能列表 + 置信度）
        """
        results = self.search.search(user_message, top_k=5)
        
        # 如果有图像文件，强制把图像模型加入候选
        if has_file and file_type and file_type.startswith("image"):
            image_results = self.search.search(
                user_message, top_k=2, categories=["model"]
            )
            results = image_results + results
        
        if not results or results[0]["score"] < self.THRESHOLD_LOW:
            return RoutingPlan(
                mode="chat_only",
                confidence="low",
                selected_resources=[]
            )
        
        if results[0]["score"] >= self.THRESHOLD_HIGH:
            return RoutingPlan(
                mode="direct_call",
                confidence="high",
                selected_resources=[results[0]]
            )
        
        # 中置信度：返回 top3，让 Claude 从中选择
        return RoutingPlan(
            mode="claude_select",
            confidence="medium",
            selected_resources=results[:3]
        )
```

```python
# backend/app/executor/skill_executor.py

import anthropic
from pathlib import Path
from app.config import settings

class SkillExecutor:
    """
    组装 skill 的 system prompt，调用 Claude API
    
    四种 skill_type 的处理逻辑：
    1. document_only → 注入 SKILL.md 到 system prompt
    2. tool_reference → 注入 SKILL.md + references 摘要
    3. executable → 注入 SKILL.md + 将 handler.py 的函数注册为 Claude tools
    4. complex_workflow → 注入 SKILL.md + 大文件摘要
    """
    
    def __init__(self):
        self.client = anthropic.AsyncAnthropic(api_key=settings.ANTHROPIC_API_KEY)
        self.skills_dir = Path(settings.SKILLS_DIR)
    
    async def execute(self, skill_metadata: dict, user_message: str, conversation_history: list = None) -> str:
        
        skill_type = skill_metadata["skill_type"]
        
        # 1. 加载 SKILL.md
        system_prompt = self._load_skill_md(skill_metadata["skill_md_path"])
        
        # 2. 根据类型追加内容
        tools = []
        
        if skill_type == "tool_reference":
            refs = self._load_references(skill_metadata["references_path"])
            system_prompt += f"\n\n## Reference Materials\n{refs}"
        
        elif skill_type == "executable":
            tools = self._load_handler_as_tools(skill_metadata["handler_path"])
        
        elif skill_type == "complex_workflow":
            # 大文件只加载摘要/前 N 行
            refs = self._load_references(skill_metadata.get("references_path"), max_chars=4000)
            system_prompt += f"\n\n## Reference Materials (truncated)\n{refs}"
        
        # 3. 调用 Claude
        messages = conversation_history or []
        messages.append({"role": "user", "content": user_message})
        
        kwargs = {
            "model": settings.CLAUDE_MODEL,
            "max_tokens": 4096,
            "system": system_prompt,
            "messages": messages
        }
        if tools:
            kwargs["tools"] = tools
        
        response = await self.client.messages.create(**kwargs)
        
        # 4. 如果 Claude 返回 tool_use，执行并继续
        if response.stop_reason == "tool_use":
            return await self._handle_tool_use(response, kwargs)
        
        return response.content[0].text
    
    def _load_skill_md(self, path: str) -> str:
        full_path = self.skills_dir / path
        return full_path.read_text(encoding="utf-8")
    
    def _load_references(self, path: str, max_chars: int = 8000) -> str:
        if not path:
            return ""
        ref_dir = self.skills_dir / path
        content = ""
        for f in sorted(ref_dir.iterdir()):
            if f.is_file() and f.suffix in [".md", ".txt", ".json"]:
                content += f"\n### {f.name}\n{f.read_text()[:max_chars]}\n"
            if len(content) > max_chars:
                break
        return content[:max_chars]
    
    def _load_handler_as_tools(self, handler_path: str) -> list[dict]:
        """
        从 handler.py 中导出的函数，转为 Claude tool_use 格式
        handler.py 应 export 一个 TOOLS 列表，每个元素是标准 tool schema
        """
        import importlib.util
        full_path = self.skills_dir / handler_path
        spec = importlib.util.spec_from_file_location("handler", full_path)
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        return getattr(module, "TOOLS", [])
```

#### 3.3.4 编排器（Orchestrator）

```python
# backend/app/executor/orchestrator.py

from app.router.route_decision import RouteDecisionMaker, RoutingPlan
from app.executor.tool_executor import ToolExecutor
from app.executor.model_executor import ModelExecutor
from app.executor.skill_executor import SkillExecutor
import anthropic
from app.config import settings

class Orchestrator:
    """
    核心编排逻辑：根据 RoutingPlan 执行调用
    
    三种模式：
    1. chat_only → 直接调用 Claude，不注入任何工具
    2. direct_call → 高置信度，直接执行选定的资源
    3. claude_select → 中置信度，把候选资源作为 tools 传给 Claude 让它选
    """
    
    def __init__(self):
        self.tool_executor = ToolExecutor()
        self.model_executor = ModelExecutor()
        self.skill_executor = SkillExecutor()
        self.claude_client = anthropic.AsyncAnthropic(api_key=settings.ANTHROPIC_API_KEY)
    
    async def run(self, user_message: str, routing_plan: RoutingPlan, 
                  file_bytes: bytes = None, filename: str = None,
                  conversation_history: list = None) -> dict:
        """
        返回: {"response": "最终回答文本", "tools_used": [...], "debug": {...}}
        """
        
        if routing_plan.mode == "chat_only":
            return await self._chat_only(user_message, conversation_history)
        
        elif routing_plan.mode == "direct_call":
            resource = routing_plan.selected_resources[0]
            return await self._direct_execute(resource, user_message, file_bytes, filename, conversation_history)
        
        elif routing_plan.mode == "claude_select":
            return await self._claude_select(routing_plan.selected_resources, user_message, file_bytes, filename, conversation_history)
    
    async def _chat_only(self, user_message: str, history: list) -> dict:
        """纯对话，不调用任何工具"""
        messages = (history or []) + [{"role": "user", "content": user_message}]
        response = await self.claude_client.messages.create(
            model=settings.CLAUDE_MODEL,
            max_tokens=4096,
            system="你是一个专业的医疗助手。如果用户的问题涉及具体的计算或诊断，建议他们提供更多信息以便调用专业工具。",
            messages=messages
        )
        return {"response": response.content[0].text, "tools_used": [], "mode": "chat_only"}
    
    async def _direct_execute(self, resource: dict, user_message: str, 
                               file_bytes: bytes, filename: str, history: list) -> dict:
        """高置信度直接执行"""
        item = resource["item"]
        category = resource["category"]
        
        if category == "tool":
            # 让 Claude 从用户消息中提取参数，然后调用工具
            return await self._execute_tool_with_claude(item, user_message, history)
        
        elif category == "model":
            if not file_bytes:
                return {"response": f"该功能需要上传{item.get('input_type', '文件')}，请上传后重试。", "tools_used": [], "mode": "direct_call"}
            result = await self.model_executor.execute(item, file_bytes, filename)
            # 让 Claude 解读模型结果
            return await self._interpret_model_result(item, result, user_message, history)
        
        elif category == "skill":
            text = await self.skill_executor.execute(item, user_message, history)
            return {"response": text, "tools_used": [item["id"]], "mode": "direct_call"}
    
    async def _execute_tool_with_claude(self, tool_item: dict, user_message: str, history: list) -> dict:
        """
        用 Claude 的 tool_use 能力从自然语言中提取参数，然后调用工具
        
        流程：
        1. 把工具的 input_schema 注册为 Claude tool
        2. 发送用户消息
        3. Claude 返回 tool_use（带提取的参数）
        4. 调用真实工具
        5. 把结果返回 Claude 生成最终回答
        """
        claude_tools = [{
            "name": tool_item["id"],
            "description": tool_item["description"],
            "input_schema": tool_item["input_schema"]
        }]
        
        messages = (history or []) + [{"role": "user", "content": user_message}]
        
        response = await self.claude_client.messages.create(
            model=settings.CLAUDE_MODEL,
            max_tokens=4096,
            system="你是医疗计算助手。请根据用户输入的信息调用合适的工具进行计算。如果用户未提供某些必需参数，请直接询问。",
            tools=claude_tools,
            messages=messages
        )
        
        if response.stop_reason == "tool_use":
            # 提取 Claude 决定的参数
            tool_use_block = next(b for b in response.content if b.type == "tool_use")
            arguments = tool_use_block.input
            
            # 调用真实 Gateway
            tool_result = await self.tool_executor.execute(tool_item, arguments)
            
            # 把结果返回给 Claude 做解读
            messages.append({"role": "assistant", "content": response.content})
            messages.append({"role": "user", "content": [{
                "type": "tool_result",
                "tool_use_id": tool_use_block.id,
                "content": str(tool_result)
            }]})
            
            final_response = await self.claude_client.messages.create(
                model=settings.CLAUDE_MODEL,
                max_tokens=4096,
                system="你是医疗计算助手。请根据工具返回的结果给用户清晰的解释。",
                tools=claude_tools,
                messages=messages
            )
            
            return {
                "response": final_response.content[0].text,
                "tools_used": [tool_item["id"]],
                "tool_result_raw": tool_result,
                "mode": "direct_call"
            }
        
        # Claude 没有调用工具（可能在追问参数）
        return {"response": response.content[0].text, "tools_used": [], "mode": "direct_call"}
```

from fastapi import APIRouter, UploadFile, File, Form

from app.schemas.chat import ChatRequest, ChatResponse

# 全局实例（应用启动时初始化）
search_engine = VectorSearchEngine()
decision_maker = RouteDecisionMaker(search_engine)
orchestrator = Orchestrator()

@router.post("/api/chat", response_model=ChatResponse)
async def chat(
    message: str = Form(...),
    file: UploadFile = File(None),
    conversation_id: str = Form(None)
):
    """
    主对话接口
    
    支持：纯文本对话 / 文本+文件（如胸片）
    """
    # 1. 路由决策
    has_file = file is not None
    file_type = file.content_type if file else None
    
    routing_plan = decision_maker.decide(
        user_message=message,
        has_file=has_file,
        file_type=file_type
    )
    
    # 2. 读取文件（如有）
    file_bytes = None
    filename = None
    if file:
        file_bytes = await file.read()
        filename = file.filename
    
    # 3. 执行
    result = await orchestrator.run(
        user_message=message,
        routing_plan=routing_plan,
        file_bytes=file_bytes,
        filename=filename,
        conversation_history=None  # TODO: 从 conversation_id 恢复历史
    )
    
    return ChatResponse(
        response=result["response"],
        tools_used=result.get("tools_used", []),
        routing_info={
            "mode": routing_plan.mode,
            "confidence": routing_plan.confidence,
            "candidates": [r["item"]["name"] for r in routing_plan.selected_resources]
        }
    )

## 八、关键技术决策汇总

| 决策项 | 选择 | 理由 |
|-------|------|------|
| 向量数据库 | FAISS (IndexFlatIP) | 数据量 ~1000 条，无需分布式，本地内存足够 |
| Embedding 模型 | pritamdeka/S-PubMedBert-MS-MARCO | 医学领域、sentence-transformers 兼容、HuggingFace 直接下载 |
| 后端框架 | FastAPI | 与已有 Gateway 统一、异步支持好 |
| LLM | Claude (Anthropic API) | tool_use 能力成熟，参数提取可靠 |
| 路由策略 | 向量检索 + 阈值分级 | 平衡准确率和延迟，高置信直接调、中置信让 Claude 选 |
| 前后端通信 | REST API (JSON) | 简单清晰，后续可加 WebSocket 做流式 |
| Skill 处理 | 按 type 分发、动态注入 system prompt | 最大化利用 Claude 能力 |

---

## 九、注意事项与约束

1. **Embedding 模型首次下载**：`pritamdeka/S-PubMedBert-MS-MARCO` 约 420MB，首次运行自动从 HuggingFace 下载，需要网络。Docker 部署时应在 build 阶段预下载。

2. **Claude API 成本控制**：`chat_only` 和 `claude_select` 模式每次对话至少 1 次 Claude API 调用；`direct_call` 工具模式是 2 次（提取参数+解释结果）。Skill 模式 1~N 次（取决于是否有 tool_use 循环）。考虑对简单工具调用在前端做参数表单收集，绕过 Claude 直接 `/execute`。

3. **Gateway 网络连通性**：MToolHub backend 容器需要能访问已有 Gateway（`GATEWAY_BASE_URL`）。Docker 部署时需在同一 network。

4. **FAISS 索引重建**：每次 registry JSON 更新后需手动运行 `python scripts/build_index.py`。后续可考虑添加 API 端点触发重建。

5. **会话管理**：Phase 1~4 暂不实现持久化会话管理（无数据库）。对话历史暂存内存。Phase 5+ 可考虑加 Redis/SQLite。

6. **安全**：医疗工具输出应始终附带免责声明。Claude system prompt 中应包含"仅供参考，不构成医疗建议"的提示。
